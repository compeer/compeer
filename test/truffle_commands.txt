var instance = await Compeer.deployed();
var accounts = await web3.eth.getAccounts();

instance.registerFunder([], "Funder 1", "Description 1", "imageUrl 1", {from: accounts[0]});

var blockNumber = await web3.eth.getBlockNumber();
var block = await web3.eth.getBlock(blockNumber);
var now = block['timestamp'];

instance.mintVestingCarrot(accounts[1], '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48', now+2000, 1, 5, now+4000, now+22000, "message 1", {from: accounts[0]});

instance.deposit(0, 100, '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48', {from: accounts[0]});
